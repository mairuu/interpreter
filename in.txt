trait Updatable { tick }

trait Describable { describe }

struct Point { x v }

impl Updatable for Point { fun tick(self) { self.x = self.x + self.v } }

impl Describable for Point {
  fun describe(self) {
    return "a point at x = " + str(self.x)
  }
}

fun simulate(u: Updatable, iteration) {
  for var i = 0; i < iteration; i = i + 1 {
    u.tick()
  }
}

simulate(Point(0, 1), 10) // auto cast
simulate(Point(0, 1) as Updatable, 10) // already at right trait
simulate(Point(0, 1) as Describable, 10) // wrong trait re-cast

struct Person { name }

var bad = Person("") as Updatable // bad, raise panic
