var ResultTrait
var Result
var Ok
var Err

// simulate import
(fun() {
  // pub { ResultTrait Result Ok Err } 

  trait std_result_ResultTrait { is_ok unwrap }
  struct std_result_Result { ok val }

  fun std_result_Ok(val) {
    return std_result_Result(true, val) as std_result_ResultTrait
  }

  fun std_result_Err(err) {
    return std_result_Result(false, err) as std_result_ResultTrait
  }

  impl std_result_ResultTrait for std_result_Result {
    fun is_ok(self) {
      return self.ok
    }

    fun unwrap(self) {
      if !self.ok {
        panic("called unwrap() on Err:", self.val)
      }
      return self.val
    }
  }

  ResultTrait = std_result_ResultTrait
  Result = std_result_Result
  Ok = std_result_Ok
  Err = std_result_Err
})()

  trait std_iterable_Iterable { next has_next }

  fun std_iterable_iterate(it: std_iterable_Iterable, cb) {
    for it.has_next() {
      cb(it.next())
    }
  }
  Iterable = std_iterable_Iterable
  iterate = std_iterable_iterate
})()

// user struct impl std trait!

struct Range { current end }

impl Iterable for Range {
  fun next(self) {
    var next = self.current
    self.current = self.current + 1
    return next
  }
  fun has_next(self) {
    return self.current < self.end
  }
}

fun divide(a, b)  {
  if b == 0 { return Err("division by zero") }
  return Ok(a / b)
}

iterate(Range(-3, 3), fun(i) {
  var r = divide(10, i)
  if r.is_ok() {
    println(r.unwrap())
  }
})
