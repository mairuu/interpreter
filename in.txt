trait Shoutable {
  shout
  inner
}

struct Echo {
  msg
}

impl Shoutable for Echo {
  fun inner(self) {
    return self.msg
  }
  fun shout(self) {
    return self.inner() + "!!!"
  }
}

var e = Echo("hello")
var s = e as Shoutable

println(s.shout()) 

trait Walkable {
  walk
}

impl Walkable for Echo {
  fun walk(self) { return self.msg + " walks" }
}

var e = Echo("hello")
var s = e as Shoutable
s.walk()  // should panic — wrong trait
